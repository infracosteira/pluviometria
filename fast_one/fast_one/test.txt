
def open_map(x, y):
    map_ = folium.Map(location=[y, x], zoom_start=12)
    folium.Marker([y, x], popup="Location").add_to(map_)
    map_.save("map.html")
    # Open the map in the default browser
    webbrowser.open("map.html")

def create_button(row):
    button = pn.widgets.Button(name="Show on Map", button_type="primary")
    button.on_click(lambda event: open_map(row['x'], row['y']))
    return button

posto['Button'] = posto.apply(create_button, axis=1)
